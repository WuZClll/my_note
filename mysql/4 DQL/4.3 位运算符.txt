-- 位运算符(了解)

select 3&5;-- 位与
/*两个都为一则是一否则为0
0011       一共32位，左面省略了
0101
---------
0001
*/

select 3|5；-- 位或
/*两个只要有一个为一结果为一
0011
0101
--------
0111
*/

select 3^5;-- 位异或
/*相同为零不同为一
0011
0101
-------
0110
*/

select 3>>1;-- 位右移，左边补零
/*
0011 >>1 ---->0001
*/

select 3<<1;-- 位左移，右边补零
/*
0011<<1 ----->0110
*/

select~3; -- 位取反
/*
0000 0000 0000 0000 0000 0000 0000 0011
----->
1111 1111 1111 1111 1111 1111 1111 1100
*/
